```c++
//接口文件
#ifndef DBDATADEF_H
#define DBDATADEF_H

//数据库访问状态
enum E_STATUS
{
    E_SUCESS = 0, //执行成功
    E_OPENDB_FAILED = 101, //数据库打开失败
    E_OPENREORDSET_FAILED, //打开数据集失败

    E_EXEC_FAILED,  //程序执行失败

//数据库信息
struct struDBInfo

    QString m_strSQLType;  //数据库类型

//信号制式
enum SignalSystem
{
    ContinuousWave,//连续波
    ImpulseWave//脉冲波
};

//类型
enum WorkType
{
    Initiative,//主动
    Passivity//被动
};

//频率类型
enum FrequencyType
{
    Stabilization,//稳定
    RandomAgility,//随机捷变
    CycleAgile//周期捷变
};

//工作频率点
enum OperatingFrequencyPoint
{
    F1,
    F2,
    F3
};

//扫描方式
enum ScanningMode
{
    Sector,//扇形
    Round,//圆形
    TrackingWhileScanning,//边扫描边跟踪
    Lock//锁定
};

//可机动地表类型
enum ManeuverableSurfaceType
{
```

```c++
/***********************************************************************
Function:    // GetConnInfo
Description: // 获取连接信息
Input:       // 无
Output:      // 无
Return:      // bool
Author:      // 马文水
Time:        // 2021-09-24
************************************************************************/
bool CDBDataManager::GetConnInfo(struDBInfo &dbInfo)
{
    QString l_qStrPath = QCoreApplication::applicationDirPath()
            + QString::fromLocal8Bit("/config/SQLConfig.ini");

    QSettings *l_pSetConfigIniRead = new QSettings(l_qStrPath, QSettings::IniFormat);
    //获取数据库类型
    dbInfo.m_strSQLType = l_pSetConfigIniRead->value("DRIVER").toString();
    //获取数据库名称
    dbInfo.m_strDatabase = l_pSetConfigIniRead->value("Database").toString();

    delete l_pSetConfigIniRead;
    l_pSetConfigIniRead = nullptr;

    return true;
}

/***********************************************************************
Function:    // ConnectDB
Description: // 连接数据库
Input:       // 无
Output:      // 无
Return:      // DBMGR::E_STATUS
Author:      // 马文水
Time:        // 2021-09-24
************************************************************************/
E_STATUS CDBDataManager::ConnectDB()
{
    //获取数据库数据
    struDBInfo l_dbInfo;
    GetConnInfo(l_dbInfo);
    //sqlite数据库需要获取文件的全局路径
    if ("sqlite" == l_dbInfo.m_strSQLType)
    {
        l_dbInfo.m_strDatabase = QCoreApplication::applicationDirPath() + "/"
                + l_dbInfo.m_strDatabase;
    }

    //若连接成功
    if (nullptr == m_pDBConn)
    {
        if (l_dbInfo.m_strSQLType.isEmpty())
        {
            m_pDBConn = new CDBProcess();
        }
        else
        {
            m_pDBConn = new CDBProcess(l_dbInfo.m_strSQLType);
        }
    }

    m_pDBConn->openDB(l_dbInfo.m_strDatabase);

    E_STATUS l_eRes = E_SUCESS;
    bool l_bRet = m_pDBConn->isOpen();
    if (!l_bRet)
    {
        //连接失败
        qDebug()<<"数据库连接失败";
        l_eRes = DBMGR::E_OPENDB_FAILED;
    }
    else
    {
        qDebug()<<"数据库连接成功";
        m_bIsConnect = true;
    }

    return l_eRes;
}

/***********************************************************************
Function:    // CloseDB
Description: // 断开数据库连接
Input:       // 无
Output:      // 无
Return:      // 无
Author:      // 马文水
Time:        // 2021-09-24
************************************************************************/
void CDBDataManager::CloseDB()
{
    if (true == m_bIsConnect)
    {
        m_pDBConn->closeDB();
        m_bIsConnect = false;
    }
}

/***********************************************************************
Function:    // getSensorInfo
Description: // 读取 “传感器” 信息
Input:       // QVector<struSensorInfo> &l_vecSensorInfo 储存 “传感器信息” 结构体的数组
Output:      // std::vector<DBMGR::struSensorInfo> &l_vecSensorInfo：传感器信息
Return:      // DBMGR::E_STATUS
Author:      // 马文水
Time:        // 2021-09-24
************************************************************************/
E_STATUS CDBDataManager::getSensorInfo(QVector<struSensorInfo> &l_vecSensorInfo)
{
    if (false == m_bIsConnect)
    {
        return E_DB_NOT_CONNECT;
    }

    QString l_strSql = QString("SELECT * from SensorList");

    void* l_pTmp = m_pDBConn->openRecordsetBySql(l_strSql);

    E_STATUS l_eRes = E_SUCESS;
    if (nullptr == l_pTmp)
    {
        l_eRes = E_OPENREORDSET_FAILED;
    }
    else
    {
        while (m_pDBConn->moveNext(l_pTmp))
        {
            QSqlQuery *l_pQry = static_cast<QSqlQuery *>(l_pTmp);

            //获取数据
            int l_iIndex = 0;
            struSensorInfo l_struSensorInfo;
            l_struSensorInfo.m_iID = l_pQry->value(l_iIndex++).toLongLong(); //ID
            l_struSensorInfo.m_strNAME = l_pQry->value(l_iIndex++).toString(); //传感器名称

            l_vecSensorInfo.append(l_struSensorInfo);
        }
        m_pDBConn->closeRecordset(l_pTmp);
    }
    return l_eRes;
}

/***********************************************************************
Function:    // addSensorInfo
Description: // 向数据库添加一条 “传感器信息” 记录
Input:       // DBMGR::struSensorInfo &l_struSensorInfo：传感器信息
Output:      // DBMGR::struSensorInfo &l_struSensorInfo：主要输出新增用户的ID
Return:      // DBMGR::E_STATUS：数据库操作返回值
Author:      // 马文水
Time:        // 2021-09-24
************************************************************************/
E_STATUS CDBDataManager::addSensorInfo(struSensorInfo &l_struSensorInfo)
{
    //若尚未连接数据库，返回
    if (false == m_bIsConnect)
    {
        return E_DB_NOT_CONNECT;
    }

    //插入语句
    QString l_strSql = "INSERT INTO SensorList (NAME) VALUES ('%1')";
    l_strSql = l_strSql.arg(l_struSensorInfo.m_strNAME);

    E_STATUS l_eRes = E_SUCESS;
    bool l_bRes = m_pDBConn->excuteSQL(l_strSql);
    if (true == l_bRes)
    {
        l_strSql.clear();
        l_strSql = QString("SELECT ID from SensorList where ID =(SELECT MAX(ID) from SensorList)");

        void *l_pTmp = m_pDBConn->openRecordsetBySql(l_strSql);

        if (nullptr == l_pTmp)
        {
            l_eRes = E_OPENREORDSET_FAILED;
        }
        else
        {
            while (m_pDBConn->moveNext(l_pTmp))
            {
                QSqlQuery *l_pQry = static_cast<QSqlQuery *>(l_pTmp);
                l_struSensorInfo.m_iID = l_pQry->value(0).toLongLong();//ID
            }
        }
    }
    else
    {
        l_eRes = E_EXEC_FAILED;
    }
    return l_eRes;
}
```

```c++
CDBProcess::CDBProcess(const QString strType /* = "sqlite"*/)
{
    __strDbType = strType.toUpper();
    QString dbType("SQLITE");
    if("SQLITE" == __strDbType)
        dbType = "QSQLITE"; //QTPLUGIN += qsqlite Q_IMPORT_PLUGIN(QSQLiteDriverPlugin)
    else if ("MYSQL" == __strDbType)
        dbType = "QMYSQL"; //QTPLUGIN += qsqlmysql Q_IMPORT_PLUGIN(QMYSQLDriverPlugin)
    else if ("SQLSERVER" == __strDbType)
        dbType = "QODBC"; //QTPLUGIN += qsqlodbc  Q_IMPORT_PLUGIN(QODBCDriverPlugin)
    else if("ACCESS" == __strDbType)
        dbType = "QODBC"; //QTPLUGIN += qsqlodbc  Q_IMPORT_PLUGIN(QODBCDriverPlugin)
    else
    {
        dbType = "";
        QMessageBox::critical(0, "ERROR", "DB type name invalid!");
        return;
    }
    int iConnIdx = 0;
    while(1)
    {
        __strConnName = QString("MyDBProcessConn%1").arg(++iConnIdx);
        QSqlDatabase dbConn = QSqlDatabase::database(__strConnName, false);

        if(dbConn.isValid())//存在连接
        {
            continue;
        }
        m_pDB = new QSqlDatabase(QSqlDatabase::addDatabase(dbType, __strConnName));
        break;
    }
}

CDBProcess::~CDBProcess()
{
    qDeleteAll(__lstQrys);

    m_pDB->close();
    delete m_pDB;
    QSqlDatabase::removeDatabase(__strConnName);
}

bool CDBProcess::openDB(const QString strDBname)
{
    return openDB("", strDBname);

bool CDBProcess::openDB(const QString strSvrName,
                        const QString strDBname,
                        const QString strUserID,
                        const QString strUserPwd)
{
    if(isOpen())
        return false;
    __strDbName = strDBname;
    bool bRet = false;
    if("SQLITE" == __strDbType)
    {
        bRet = __openSqlite(strDBname);
    }
    else if("MYSQL" == __strDbType)
    {
        bRet = __openMySql(strSvrName, strDBname, strUserID, strUserPwd);
    }
    else if("SQLSERVER" == __strDbType)
    {
        bRet = __connectSqlServer(strSvrName, strDBname, strUserID, strUserPwd);
    }
    else if("ACCESS" == __strDbType)
    {
        if(strSvrName.length() < 1)
        {
            bRet = __openMDB(strDBname, strUserID, strUserPwd);
        }
        else
        {
            bRet = __openMDBByUDL(strSvrName); //here strSvrName is UDL file name
        }
    }

bool CDBProcess::__openMDBByUDL(const QString strUDL)
{
    QFile file(strUDL);
    if(!file.open(QIODevice::ReadOnly)) return false;

    QTextStream in(&file);
    in.setCodec("UTF-8");
    QString strAll = in.readAll();
    if(strAll.length() > 3 && QChar(0) == strAll[1])
    {
        in.setCodec("UTF-16");
        in.seek(0);
        strAll = in.readAll();
    }
    file.close();
    int idx1 = strAll.indexOf("=");
    idx1 = strAll.indexOf("=", idx1 + 1);
    int idx2 = strAll.indexOf(";", idx1 + 1);
    QString strMDB = strAll.mid(idx1 + 1, idx2 - idx1 - 1);
    strMDB = strMDB.trimmed();

    if(QFileInfo(strMDB).fileName() == strMDB)
    {
        strMDB = QString("%1/%2").arg( QCoreApplication::applicationDirPath() ).arg( strMDB );
    }
    return __openMDB(strMDB);
```

```c++
    if(ui->treeWidget->currentItem()->isSelected())
    {

        // "确认删除" 弹窗

        mymsgbox->setIcon(QMessageBox::Warning);
        mymsgbox->setWindowTitle("提示");
        mymsgbox->setText("确定删除？");
        mymsgbox->addButton(determinePushButton, QMessageBox::AcceptRole);
        mymsgbox->addButton(cancelPushButton, QMessageBox::RejectRole);
        mymsgbox->show();

        mymsgbox->exec();//阻塞等待用户操作

        if(mymsgbox->clickedButton() == determinePushButton)
        {
            //删除数据库中的数据
            CDBDataManager::getInstance()->deleteSensorInfo(m_strSelectedChildText.toInt());
            //删除QTreeWidget节点
            removeSelectedItems(ui->treeWidget);
        }
    }
    else
    {

    }
}

void Widget::removeItem(QTreeWidgetItem *pTreeItem)
{
    int l_icount = pTreeItem->childCount();//子节点数
    if(l_icount==0)//没有子节点，直接删除
    {
        delete pTreeItem;
        return;
    }

    for(int i=0; i < l_icount; i++)
    {
        QTreeWidgetItem *childItem = pTreeItem->child(0);//删除子节点
        removeItem(childItem);
    }
    delete pTreeItem;//最后将自己删除
}

Function:    // slotFinishEdit
Description: // 编辑完成后,移除widget,还原树对应的槽函数
Input:       // 无
Output:      // 无
Return:      // 无
Author:      // 马文水
Time:        // 2021-10-11
************************************************************************/
void Widget::slotFinishEdit()
{
    QLineEdit *pEdit=qobject_cast<QLineEdit*>(ui->treeWidget->itemWidget(ui->treeWidget->currentItem(),0));

    if(!pEdit)
    {
        return;
    }

    //获取新名称
    QString l_strNewName=pEdit->text();

    //判断名称是否改变
    if(m_strOldName == l_strNewName)
    {
        ui->treeWidget->removeItemWidget(ui->treeWidget->currentItem(),0);
        return;
    }

    //若名称已改变，判断命名是否重复
    if(CDBDataManager::getInstance()->isExistsSensor(l_strNewName))
    {
        QMessageBox::warning(this,"重命名失败","该名称已存在","确定");
        ui->treeWidget->removeItemWidget(ui->treeWidget->currentItem(),0);
        return;
    }

    // 移除widget
    ui->treeWidget->removeItemWidget(ui->treeWidget->currentItem(),0);
    // 赋值
    ui->treeWidget->currentItem()->setText(0,l_strNewName);
    //更新数据库中的数据
    struSensorInfo l_struSensorInfo;
    l_struSensorInfo.m_iID = m_strSelectedChildText.toInt();
    l_struSensorInfo.m_strNAME.append(l_strNewName);
    CDBDataManager::getInstance()->updateSensorInfo(l_struSensorInfo);
}

/***********************************************************************
Function:    // on_preservationPushButton_clicked
Description: // 按下保存按钮后获取右侧界面数据保存到对应数据库中
Input:       // 无
Output:      // 无
Return:      // 无
Author:      // 马文水
Time:        // 2021-09-30
************************************************************************/
void Widget::on_preservationPushButton_clicked()
{
    //判断选中内容是否为空
    if(ui->treeWidget->currentItem() == nullptr)
    {
        QMessageBox::warning(this,"提示","请选择传感器","确定");
        return;
    }

    // "确认保存" 弹窗
    QMessageBox *pConfirmSaveBox = new QMessageBox;
    QPushButton *pDeterminePushButton = new QPushButton("确定");
    QPushButton *pCancelPushButton = new QPushButton("取消");

    pConfirmSaveBox->setIcon(QMessageBox::Warning);
    pConfirmSaveBox->setWindowTitle("提示");
    pConfirmSaveBox->setText("确定保存？");
    pConfirmSaveBox->addButton(pDeterminePushButton, QMessageBox::AcceptRole);
    pConfirmSaveBox->addButton(pCancelPushButton, QMessageBox::RejectRole);
    pConfirmSaveBox->show();

    pConfirmSaveBox->exec();//阻塞等待用户操作

    if(pConfirmSaveBox->clickedButton() == pDeterminePushButton)
    {
        //判断更改数据所属的表
        if(m_bTechnicalParameterChanged == true)
        {
            struTechnicalParameter l_struTechnicalParameter;
            //获取界面上的数据
            l_struTechnicalParameter.m_iID = m_strSelectedChildText.toInt();//ID
            l_struTechnicalParameter.m_strOperatingFrequency = ui->OperatingFrequencyLineEdit->text();//工作频率
            l_struTechnicalParameter.m_strOperatingPower = ui->OperatingPowerLineEdit->text();//工作功率
            l_struTechnicalParameter.m_emSignalSystem = (DBMGR::SignalSystem)ui->SignalSystemComboBox->currentIndex();//信号制式
            l_struTechnicalParameter.m_emWorkType = (DBMGR::WorkType)ui->WorkTypeComboBox->currentIndex(); //类型
            l_struTechnicalParameter.m_dbScopeOfExploration = ui->ScopeOfExplorationLineEdit->text(); //探测范围
            l_struTechnicalParameter.m_dbAzimuthMin = ui->AzimuthMinLineEdit->text().toDouble(); //方位角最小值
            l_struTechnicalParameter.m_dbAzimuthMax = ui->AzimuthMaxLineEdit->text().toDouble(); //方位角最大值
            l_struTechnicalParameter.m_dbAngularAltitudeMin = ui->AngularAltitudeMinLineEdit->text().toDouble(); //高低角最小值
            l_struTechnicalParameter.m_dbAngularAltitudeMax = ui->AngularAltitudeMaxLineEdit->text().toDouble(); //高低角最大值
            l_struTechnicalParameter.m_strElevationMin = ui->ElevationMinLineEdit->text(); //高程最小值
            l_struTechnicalParameter.m_strElevationMax = ui->ElevationMaxLineEdit->text(); //高程最大值
            //保存到数据库
            CDBDataManager::getInstance()->updateTechnicalParameterInfo(l_struTechnicalParameter);
        }

        //判断更改数据所属的表
        if(m_bWorkPatternChanged == true)
        {
            struWorkPattern l_struWorkPattern;
            //获取界面上的数据
            l_struWorkPattern.m_iID = m_strSelectedChildText.toInt(); //ID
            l_struWorkPattern.m_dbFrequencySweepPeriod = ui->FrequencySweepPeriodSpinBox->text().toDouble(); //频率扫描周期
            l_struWorkPattern.m_dbFrequencyStability = ui->FrequencyStabilitySpinBox->text().toDouble(); //频率稳定性
            l_struWorkPattern.m_emFrequencyType = (DBMGR::FrequencyType)ui->FrequencyTypeComboBox->currentIndex(); //频率类型
            l_struWorkPattern.m_emOperatingFrequencyPoint = (DBMGR::OperatingFrequencyPoint)ui->OperatingFrequencyPointComboBox->currentIndex(); //工作频率点
            l_struWorkPattern.m_dbWorkingHoursMax = ui->WorkingHoursMaxLineEdit->text().toDouble(); //最大工作时长
            l_struWorkPattern.m_dbWorkingHoursMin = ui->WorkingHoursMinLineEdit->text().toDouble(); //最小工作时长
            l_struWorkPattern.m_dbDurationOfWorkingModeSwitchover = ui->DurationOfWorkingModeSwitchoverLineEdit->text().toDouble(); //工作模式切换时长
            l_struWorkPattern.m_emScanningMode = (DBMGR::ScanningMode)ui->ScanningModeComboBox->currentIndex(); //扫描方式
            //保存到数据库
            CDBDataManager::getInstance()->updateWorkPatternInfo(l_struWorkPattern);
        }

        //判断更改数据所属的表
        if(m_bTacticsApplicationChanged == true)
        {
            struTacticsApplication l_struTacticsApplication;
            //获取界面上的数据
            l_struTacticsApplication.m_iID = m_strSelectedChildText.toInt(); //ID

            QStringList l_listCheckBoxSelection;
            CheckBoxSelection(l_listCheckBoxSelection);
            //将数组拼接成字符串
            QString l_strManeuverableSurfaceType;
            for (int i = 0; i < l_listCheckBoxSelection.size(); ++i) {
                if(i == 0)
                {
                    l_strManeuverableSurfaceType = l_strManeuverableSurfaceType + l_listCheckBoxSelection[i];
                }
                else
                {
                    l_strManeuverableSurfaceType = l_strManeuverableSurfaceType + "," + l_listCheckBoxSelection[i];
                }
            }
            l_struTacticsApplication.m_strManeuverableSurfaceType = l_strManeuverableSurfaceType; //可机动地表类型
            l_struTacticsApplication.m_emSurfaceFlatness = (DBMGR::SurfaceFlatness)ui->SurfaceFlatness->currentIndex(); //地表平整度
            l_struTacticsApplication.m_dbGradient = ui->GradientLineEdit->text().toDouble(); //爬坡度
            l_struTacticsApplication.m_strConstructionAreaOf = ui->ConstructionAreaOfLineEdit->text(); //布设面积
            l_struTacticsApplication.m_emSurfaceHardness = (DBMGR::SurfaceHardness)ui->SurfaceHardness->currentIndex(); //地表硬度
            l_struTacticsApplication.m_bAllWeatherSupport = ui->AllWeatherSupport->currentIndex(); //是否支持全天时全天候
            l_struTacticsApplication.m_emAntijammingCapability = (DBMGR::AntijammingCapability)ui->AntijammingCapability->currentIndex(); //抗干扰能力
            l_struTacticsApplication.m_bDetectionOfMovingTarget = ui->DetectionOfMovingTarget->currentIndex(); //支持探测动目标
            l_struTacticsApplication.m_emModeOfDelivery = (DBMGR::ModeOfDelivery)ui->ModeOfDelivery->currentIndex(); //运载方式
            //保存到数据库
            CDBDataManager::getInstance()->updateTacticsApplicationInfo(l_struTacticsApplication);
        }
    }
}

/***********************************************************************
Function:    // CheckBoxSelection
Description: // 获取勾选的内容
Input:       // QStringList &l_listCheckBoxSelection
Output:      // 无
Return:      // 无
Author:      // 马文水
Time:        // 2021-09-30
************************************************************************/
void Widget::CheckBoxSelection(QStringList &l_listCheckBoxSelection)
{
    if(ui->PloughCheckBox->checkState())
    {
        l_listCheckBoxSelection.push_back(QString::number(DBMGR::ManeuverableSurfaceType::Plough));//耕地
    }
    if(ui->ForestCheckBox->checkState())
    {
        l_listCheckBoxSelection.append(QString::number(DBMGR::ManeuverableSurfaceType::Forest));//森林
    }
    if(ui->LawnCheckBox->checkState())
    {
        l_listCheckBoxSelection.append(QString::number(DBMGR::ManeuverableSurfaceType::Lawn));//草地
    }
    if(ui->ShrublandCheckBox->checkState())
    {
        l_listCheckBoxSelection.append(QString::number(DBMGR::ManeuverableSurfaceType::Shrubland));//灌木地
    }
    if(ui->WetlandCheckBox->checkState())
    {
        l_listCheckBoxSelection.append(QString::number(DBMGR::ManeuverableSurfaceType::Wetland));//湿地
    }
    if(ui->WaveCheckBox->checkState())
    {
        l_listCheckBoxSelection.append(QString::number(DBMGR::ManeuverableSurfaceType::Wave));//水体
    }
    if(ui->TundraCheckBox->checkState())
    {
        l_listCheckBoxSelection.append(QString::number(DBMGR::ManeuverableSurfaceType::Tundra));//苔原
    }
    if(ui->ArtificialSurfaceCheckBox->checkState())
    {
        l_listCheckBoxSelection.append(QString::number(DBMGR::ManeuverableSurfaceType::ArtificialSurface));//人造地表
    }
    if(ui->BareLandCheckBox->checkState())
    {
        l_listCheckBoxSelection.append(QString::number(DBMGR::ManeuverableSurfaceType::BareLand));//裸地
    }
    if(ui->GlaciersAndPermanentSnowCoverCheckBox->checkState())
    {
        l_listCheckBoxSelection.append(QString::number(DBMGR::ManeuverableSurfaceType::GlaciersAndPermanentSnowCover));//冰川和永久积雪
    }
}
```
